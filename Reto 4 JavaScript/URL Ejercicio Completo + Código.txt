Codigo JS:



var fondo = document.querySelector("body")
var botonColor = document.querySelector("button#color")
var reseñasOcultas = document.querySelector("div.hide")
var botonVerMas = document.querySelector("button.vermas")

function cambioColor(){
  if(fondo.classList.contains("oscuro")){
  fondo.classList.remove("oscuro")
  fondo.classList.add("claro")
  }
  else
    fondo.classList.add("oscuro")
}

function verMas(){
  botonVerMas.textContent = (botonVerMas.textContent=="ver más")?"ver menos":"ver más"
  reseñasOcultas.classList.toggle("hide")
}

botonColor.addEventListener("click", cambioColor)
botonVerMas.addEventListener("click", verMas)
